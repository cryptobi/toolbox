macro(toolbox_prog fn)
	add_executable(${fn} "${fn}.cpp")
	target_include_directories(${fn} PUBLIC ${include_dirs})
	target_link_libraries(${fn} cbtoolbox-static)
endmacro()

toolbox_prog("blocks_to_sql")
